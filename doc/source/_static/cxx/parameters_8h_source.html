384 inline void setCalculatedValues(SimulationParameters& par)
385 {
386  par.electrostaticPrefactor = par.elementaryCharge /(4.0 * M_PI * par.dielectricConstant * par.permittivitySpace * par.gridFactor);
387  par.inverseKT = par.elementaryCharge /(par.boltzmannConstant * par.temperatureKelvin);
388  if (par.outputWidth < 8 ) { par.outputWidth = 8; }
389  if (par.outputWidth < (par.outputPrecision + 8)) { par.outputWidth = par.outputPrecision + 8; }
390 }
